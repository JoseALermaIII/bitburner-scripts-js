import * as getServers from "getServers.js";
/** @param {NS} ns **/

export async function main(ns) {
	let servers = await getServers.main(ns);
	let server = null;
	if (ns.args.length === 0) {
		server = servers.get("n00dles");
	} else {
		server = servers.get(ns.args[0]);
	}
	let hostname = server.details["hostname"];
	let moneyAvailable = server.details["moneyAvailable"];
	let moneyMax = server.details["moneyMax"];
	ns.tprint(`${hostname} stats:`);
	ns.tprint(`maxRam: ${server.details["maxRam"]}`);
	ns.tprint(`cpuCores: ${server.details["cpuCores"]}`);
	ns.tprint(`serverGrowth: ${server.details["serverGrowth"]}`);
	ns.tprint(`moneyMax: ${moneyMax}`);
	ns.tprint(`minDifficulty: ${server.details["minDifficulty"]}`);
	ns.tprint(`hackDifficulty: ${server.details["hackDifficulty"]}`);
	let numThreads = null;
	let hackAmount= null;
	if (moneyAvailable < moneyMax) {
		numThreads = ns.hackAnalyzeThreads(hostname, moneyAvailable);
		hackAmount = moneyAvailable;
	} else {
		numThreads = ns.hackAnalyzeThreads(hostname, moneyMax);
		hackAmount = moneyMax;
	}
	ns.tprint(`Number of threads to get ${hackAmount} from ${hostname}: ${numThreads}`);

}
